ALTER TABLE COMPRAS ADD FATURADO CHAR(1);
ALTER TABLE parametros ADD IMPORTAR_OPERADOR_COMPRA CHAR(1);

DROP TRIGGER IF EXISTS `crm_sgr`.`compras_AFTER_INSERT`;

DELIMITER $$
USE `crm_sgr`$$
CREATE DEFINER = CURRENT_USER TRIGGER `crm_sgr`.`compras_AFTER_INSERT` 
 AFTER INSERT ON `compras` FOR EACH ROW
BEGIN
  if (new.FATURADO = 'S' and new.OPERADOR > 0) then
  
    UPDATE CLIENTES SET OPERADOR = new.OPERADOR
    where CODIGO = new.CLIENTE and ATIVO = 'SIM';
  
  end if;
  
END$$
DELIMITER ;
